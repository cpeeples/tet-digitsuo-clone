77static float bilinear_interpolate(const DrawGrid *grid, float src_x, float src_y)
78{
79 int x0 = (int)src_x;
80 int y0 = (int)src_y;
81 float fx = src_x - x0;
82 float fy = src_y - y0;
83
84 if (x0 < 0 || x0 >= GRID_SIZE - 1 || y0 < 0 || y0 >= GRID_SIZE - 1)
85 {
86 return 0.0f;
87 }
88
89 float v00 = grid->cells[y0][x0];
90 float v01 = grid->cells[y0][x0 + 1];
91 float v10 = grid->cells[y0 + 1][x0];
92 float v11 = grid->cells[y0 + 1][x0 + 1];
93
94 return (1 - fx) * (1 - fy) * v00 + fx * (1 - fy) * v01 + (1 - fx) * fy * v10 + fx * fy * v11;
95}
